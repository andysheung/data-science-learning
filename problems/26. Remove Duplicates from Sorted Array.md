class Solution:

def removeDuplicates(self, nums: List[int]) -> int:

check_set = set()

nums1 = nums

j = 0 # j loops through nums1

k = 0 #counter

for i in range(len(nums)): # i loops through nums

if nums[i] not in check_set:

nums1[j] = nums[i]

check_set.add(nums1[j])

j+=1

k+=1

nums = nums1

return k
